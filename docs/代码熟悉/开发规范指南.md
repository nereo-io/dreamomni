### 1.1 TypeScript 规范

```json
// tsconfig.json
{
```

## 二、架构设计规范

### 2.1 分层架构

#### 三层架构模式
```
┌─────────────┐
│  Component  │ ← UI层：React组件，处理用户交互
├─────────────┤
│   Service   │ ← 业务层：业务逻辑，数据处理
├─────────────┤
│    Model    │ ← 数据层：数据库操作，API调用
└─────────────┘
```

#### 层级职责划分

**Model层 (`/models`)**
```typescript
// models/user.ts
export async function findUserByEmail(email: string): Promise<User | undefined> {
  const supabase = getSupabaseClient();
  const { data, error } = await supabase
    .from("users")
    .select("*")
    .eq("email", email)
    .single();
  
  return error ? undefined : data;
}
```

**Service层 (`/services`)**
```typescript
// services/creditsService.ts
export class CreditsService {

    }
    // ...
  }
}
```


### 2.2 API 路由规范

#### RESTful 设计
```typescript
// app/api/video-generation/submit/route.ts
export async function POST(request: Request) {
  // 1. 请求验证
  const session = await auth();
  if (!session) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }
  
  // 2. 参数验证
  const body = await request.json();
  const validated = videoGenerationSchema.parse(body);
  
  // 3. 业务处理
  try {
    const result = await videoService.generate(validated);
    return NextResponse.json(result);
  } catch (error) {
    // 4. 错误处理
    return NextResponse.json(
      { error: error.message },
      { status: 400 }
    );
  }
}
```

### 2.3 错误处理规范

#### 统一错误响应格式
```typescript
interface ErrorResponse {
  error: {
    code: string;        // 错误代码
    message: string;     // 用户友好的消息
    details?: any;       // 详细信息（开发环境）
  };
  timestamp: string;
  path: string;
}

// 示例实现
export function handleApiError(error: any, path: string): NextResponse {
  const isDev = process.env.NODE_ENV === 'development';
  
  return NextResponse.json({
    error: {
      code: error.code || 'INTERNAL_ERROR',
      message: error.message || 'An error occurred',
      details: isDev ? error.stack : undefined
    },
    timestamp: new Date().toISOString(),
    path
  }, {
    status: error.status || 500
  });
}
```

#### 业务错误类
```typescript
// lib/errors.ts
export class BusinessError extends Error {
  constructor(
    message: string,
    public code: string,
    public status: number = 400
  ) {
    super(message);
    this.name = 'BusinessError';
  }
}

// 使用示例
throw new BusinessError('Insufficient credits', 'INSUFFICIENT_CREDITS', 402);
```

## 三、数据库规范

### 3.1 数据模型设计

#### 命名规范
- 表名：复数形式，snake_case (如 `video_generations`)
- 字段名：snake_case (如 `created_at`)
- 主键：`id` (UUID)
- 外键：`表名_id` (如 `user_id`)

#### 时间字段标准
```sql
created_at TIMESTAMPTZ DEFAULT NOW(),
updated_at TIMESTAMPTZ DEFAULT NOW()
```

### 3.2 Row Level Security (RLS)

#### RLS策略示例
```sql
-- 用户只能查看自己的数据
CREATE POLICY "Users can view own data" ON video_generations
  FOR SELECT
  USING (auth.uid() = user_id);

-- 用户只能更新自己的数据
CREATE POLICY "Users can update own data" ON video_generations
  FOR UPDATE
  USING (auth.uid() = user_id);
```

```typescript
// 记录日志时过滤敏感信息
function sanitizeLog(data: any) {
  const sensitive = ['password', 'token', 'apiKey', 'secret'];
  const sanitized = { ...data };
  
  sensitive.forEach(key => {
    if (sanitized[key]) {
      sanitized[key] = '***';
    }
  });
  
  return sanitized;
}
```
