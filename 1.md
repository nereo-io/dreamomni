-- =====================================================
-- 将用户 f54371bb-72c1-4c41-91ef-3c9c0b924704 最近的 10 张图片标记为精选
-- =====================================================

-- 先查看该用户的最近 10 张图片（确认数据）
SELECT
id,
prompt,
mode,
status,
created_at,
CASE
WHEN image_urls_r2 IS NOT NULL THEN 'R2'
WHEN image_urls IS NOT NULL THEN 'URLs'
ELSE 'No Image'
END as image_storage
FROM image_generations
WHERE
user_id = 'f54371bb-72c1-4c41-91ef-3c9c0b924704'
AND status IN ('COMPLETED', 'SAVED_TO_R2')
AND is_delete = false
AND (image_urls IS NOT NULL OR image_urls_r2 IS NOT NULL)
ORDER BY created_at DESC
LIMIT 10;

-- 标记为精选
WITH user_recent_images AS (
SELECT
id,
ROW_NUMBER() OVER (ORDER BY created_at DESC) as rn
FROM image_generations
WHERE
user_id = 'f54371bb-72c1-4c41-91ef-3c9c0b924704'
AND status IN ('COMPLETED', 'SAVED_TO_R2')
AND is_delete = false
AND (image_urls IS NOT NULL OR image_urls_r2 IS NOT NULL)
ORDER BY created_at DESC
LIMIT 10
)
UPDATE image_generations
SET
is_featured = true,
featured_at = NOW(),
featured_order = (SELECT rn FROM user_recent_images WHERE user_recent_images.id = image_generations.id),
featured_by = 'admin'
WHERE id IN (SELECT id FROM user_recent_images);

-- 验证更新结果
SELECT
id,
user_id,
prompt,
mode,
is_featured,
featured_at,
featured_order,
created_at
FROM image_generations
WHERE
user_id = 'f54371bb-72c1-4c41-91ef-3c9c0b924704'
AND is_featured = true
ORDER BY featured_order;

-- 查看所有精选图片的统计
SELECT
COUNT(\*) as total_featured,
COUNT(DISTINCT user_id) as unique_users,
MIN(featured_order) as min_order,
MAX(featured_order) as max_order
FROM image_generations
WHERE is_featured = true;
